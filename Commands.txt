#mat2
mat2 arquivo.jpg              # remove metadados e cria 'arquivo.cleaned.jpg'
mat2 -s arquivo.jpg           # mostra os metadados encontrados, sem alterar o arquivo
mat2 -l arquivo.jpg           # lista apenas os metadados que podem ser removidos
mat2 --inplace arquivo.jpg    # remove metadados sobrescrevendo o arquivo original
mat2 -o saida.jpg arquivo.jpg # remove metadados e salva como 'saida.jpg'
mat2 *.jpg                    # limpa metadados de todos os arquivos .jpg na pasta
mat2 -L                       # lista todos os formatos suportados pelo mat2
mat2 -v arquivo.jpg           # modo verbose (mostra detalhes da operação)
mat2 -vv arquivo.jpg          # modo mais detalhado ainda
mat2 -q arquivo.jpg           # modo quiet (só mostra erros)
mat2 -V                       # mostra a versão do mat2
mat2 -h                       # mostra a ajuda (manual de uso)

#TransShell
trans "hello world" #Traduz automaticamente do inglês para o idioma padrão do sistema
trans en:pt "house" #Tradução de inglês (en) para português (pt)
trans -b en:pt "good morning" #Mostra só a tradução, sem explicações
trans -d en:pt house #Mostra significado, sinônimos, frases exemplo
trans -p en:pt "good night" #Mostra tradução + toca o áudio da pronúncia (Google TTS)
trans -sp en: "good night" #Só fala o texto, sem mostrar tradução
trans -i arquivo.txt -o traduzido.txt ru:en #Traduz o conteúdo de um arquivo e salva em outro
trans -I #Abre um prompt onde você pode digitar frases e ir traduzindo em tempo real
trans :pt "guten tag" #: sem origem faz o trans detectar automaticamente o idioma
trans -play "good night" #Fala a frase traduzida automaticamente
trans "texto" #tradução rápida
trans en:pt "texto" #origem/destino fixo
trans -b #só a tradução
trans -d #dicionário completo
trans -p #com pronúncia
trans -i arquivo.txt → lê o conteúdo do arquivo de entrada
trans -o traduzido.txt → salva a tradução no arquivo de saída
